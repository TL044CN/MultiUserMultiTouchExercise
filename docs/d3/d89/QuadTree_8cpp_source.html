80  // DEBUGFN(1);
81  if ( quadrant.area.width <= 1 || quadrant.area.height <= 1 ) return 4;
82 
83  const auto pivotX = quadrant.area.x + (quadrant.area.width / 2);
84  const auto pivotY = quadrant.area.y + (quadrant.area.height / 2);
85 
86  uint8_t index = 0b00;
87  index |= (point.x > pivotX) ? 0b01 : 0b00;
88  index |= (point.y > pivotY) ? 0b10 : 0b00;

92  );
93  */
94

127 QuadTree::findLeaf(const std::unique_ptr<Finger>& finger, bool createLeaf) {
128  std::stack<uniqueRef<Quadrant>> quadStack;
129  uniqueRef<Quadrant> currentQuadrant = std::ref(mRoot);
130  quadStack.push(currentQuadrant);
131 
132  bool leaf = false;
133 
134  while ( currentQuadrant.get()->quadrants && !currentQuadrant.get()->data ) {
135  uint8_t subindex = findSubquadrantIndex(currentQuadrant.get()->area, finger->lastPosition());
136 
137  // last quadrant found.
138  if ( !currentQuadrant.get()->quadrants->at(subindex) ) {
139  if ( createLeaf ) {
140  if ( currentQuadrant.get()->data ) break;
141  subdivide(*currentQuadrant.get(), subindex);
142  currentQuadrant = std::ref(currentQuadrant.get()->quadrants->at(subindex));
143  quadStack.push(currentQuadrant);
144  // the new node is definitely a leaf.
145  leaf = true;
146  }
147  break;
148  }
149 
150  currentQuadrant = std::ref(currentQuadrant.get()->quadrants->at(subindex));
151  quadStack.push(currentQuadrant);
152  }
153 
154  // if the node holds data it is a leaf. alternatively a new node is also a leaf
155  if ( currentQuadrant.get()->data ) leaf = true;
156  return { quadStack, leaf };
157 }

163 void QuadTree::insert(std::unique_ptr<Finger>&& finger) {
164  // DEBUGFN(1);
165  // first find the node the Finger should go into
166  auto [stack, leaf] = findLeaf(finger, true);
167 
168  auto& leafNode = stack.top();
169 
170  if ( leafNode.get()->data ) {
171  // the node already holds data. the data has to be moved.
172  std::unique_ptr<Finger> oldData = std::move(leafNode.get()->data);
173  uint8_t oldSubIndex = findSubquadrantIndex(*leafNode.get(), oldData->lastPosition());
174  uint8_t newSubIndex = findSubquadrantIndex(*leafNode.get(), finger->lastPosition());
175 
176  while ( oldSubIndex == newSubIndex ) {
177 
178  if ( oldSubIndex >= 4 || newSubIndex >= 4 ) {
179  DEBUGFNEX("Impossible SubIndex for: old[%lu][%f, %f] new[%lu][%f, %f]",
180  oldData->id(), oldData->lastPosition().x, oldData->lastPosition().y,
181  finger->id(), finger->lastPosition().x, finger->lastPosition().y
182  );
183  throw "Impossible Subquads";
184  }
185 
186  subdivide(*leafNode.get(), oldSubIndex);
187  leafNode = std::ref(leafNode.get()->quadrants->at(oldSubIndex));
188  oldSubIndex = findSubquadrantIndex(*leafNode.get(), oldData->lastPosition());
189  newSubIndex = findSubquadrantIndex(*leafNode.get(), finger->lastPosition());
190  }
191 
192  subdivide(*leafNode.get(), oldSubIndex);
193  subdivide(*leafNode.get(), newSubIndex);
194 
195  leafNode.get()->quadrants->at(oldSubIndex)->data = std::move(oldData);
196  leafNode.get()->quadrants->at(newSubIndex)->data = std::move(finger);
197 
198  // backtracking should not be needed, because no unneccessary leaf nodes should be generated
199  } else {
200  // leaf doesnt hold data. insert directly
201  leafNode.get()->data = std::move(finger);
202  }
203  ++mSize;
204 
205 }

207 bool QuadTree::remove(const uniqueRef<Finger>& fingerRef) {
208  DEBUGFN(1);
209  auto& finger = fingerRef.get();
210  Quadrant* currentQuadrant = mRoot.get();
211  std::pair<Quadrant*, uint8_t> parentRelationship = { nullptr, 4 };
212 
213  while ( currentQuadrant->quadrants ) {
214  uint8_t subQuadIndex = findSubquadrantIndex(*currentQuadrant, finger->lastPosition());
215  DEBUGFNEX("2.1 Traversing to Subquad %i", subQuadIndex);
216  // finger not in quad tree.
217  if ( !currentQuadrant->quadrants->at(subQuadIndex) ) {
218  DEBUGFNEX("2.2 No Quadrants found -> there can't be any data here.");
219  return false;
220  }
221  parentRelationship = { currentQuadrant, subQuadIndex };
222  currentQuadrant = currentQuadrant->quadrants->at(subQuadIndex).get();
223  }
224 
225  DEBUGFNEX("3 Checking Data...");
226  // finger not in quad tree
227  if ( ! currentQuadrant->data ) return false;
228  DEBUGFNEX("3.1 Data found");
229  if ( *currentQuadrant->data != *finger ) return false;
230  DEBUGFNEX("3.2 Data holds the Finger");
231 
232  // finger found. remove it
233  currentQuadrant->data.reset();
234  // remove the quadrant
235  parentRelationship.first->quadrants->at(parentRelationship.second).reset();
236 
237  --mSize;
238  return true;

241 void QuadTree::update(const uniqueRef<Finger>& fingerRef, const cv::RotatedRect& ellipse) {
242  auto& finger = fingerRef.get();
243 
244  DEBUGFNEX("Finding Leaf Node for given Finger");
245  auto [quadStack, leaf] = findLeaf(finger, true);
246  auto& leafNode = quadStack.top();
247  if ( !leafNode.get()->data ) {
248  DEBUGFNEX("Finger is not part of the Quad Tree. inserting...");
249  leafNode.get()->data = std::move(finger);
250  return;
251  } else {
252  DEBUGFNEX("Leaf with Data found. comparing ID's");
253  if ( *leafNode.get()->data != *finger ) {
254  DEBUGFNEX("Mismatching ID's! throwing away Finger to update...");
255  finger.reset();
256  return;
257  }
258 
259  // this should be the same as the unique_ptr referenced by fingerRef,
260  // but to make sure it is moved, move the unique_ptr to the context of this function
261  DEBUGFNEX("Updating Finger properties and storing in local scope");
262  std::unique_ptr<Finger> fingerData = std::move(leafNode.get()->data);
263  fingerData->addNewPosition(ellipse);
264 
265  DEBUGFNEX("Cleaning up unused Nodes...");
266 
267  // cleanup unused nodes
268  uint8_t freeCount = 0;
269  while ( !quadStack.empty() && freeCount < 4 ) {
270  freeCount = 0;
271  Quadrant* parentQuad = quadStack.top().get().get();
272  quadStack.pop();
273 
274  DEBUGFNEX("Check Quadrant: [%i , %i]", parentQuad->area.x, parentQuad->area.y);
275 
276  if ( !parentQuad->quadrants ) {
277  DEBUGFNEX("Quadrant has no subquadrants.");
278  continue;
279  }
280 
281  for ( auto& quadrant : *parentQuad->quadrants ) {
282  if ( !quadrant ) {
283  freeCount++;
284  continue;
285  }
286  if ( !quadrant->data && !quadrant->quadrants ) {
287  DEBUGFNEX("Quadrant [%i , %i] has no data or subquadrants. remove it.", quadrant->area.x, quadrant->area.y);
288  quadrant.reset();
289  freeCount++;
290  }
291  }
292 
293  if ( freeCount == 4 ) {
294  DEBUGFNEX("Completely unused Quadrant... removing all Quadrants");
295  parentQuad->quadrants.reset();
296  } else {
297  DEBUGFNEX("Free Quadrants: %i", freeCount);
298  }
299  }
300 
301  // insert the updated Finger Data back into the quad tree
302  insert(std::move(fingerData));
303 
304  }

308 std::vector<uniqueRef<Finger>> QuadTree::intersect(const cv::Point2f& center, double radius) const {
309  // DEBUGFN(1);
310  std::vector<uniqueRef<Finger>> result;
311  std::stack<Quadrant*> quadStack;
312 
313  quadStack.push(mRoot.get());
314  while ( !quadStack.empty() ) {
315  Quadrant* currentQuad = quadStack.top();
316  quadStack.pop();
317 
318  if ( currentQuad->quadrants ) {
319  // DEBUGFNEX(2,".1 Check Quadrants");
320  for ( auto& quadrant : *currentQuad->quadrants ) {
321  if ( !quadrant ) continue;
322  if ( !intersectRectCircle(quadrant->area, center, radius) ) continue;
323  // DEBUGFNEX(2, ".2 Quadrant intersects");
324  quadStack.push(quadrant.get());
325  }
326  }
327 
328  if ( currentQuad->data ) {
329  // DEBUGFNEX(3, ".1 Quadrant holds Data");
330  if ( inCircle(currentQuad->data->lastPosition(), center, radius) ) {
331  // DEBUGFNEX(3, ".2 Data lies in Radius");
332  result.push_back(std::ref(currentQuad->data));
333  }
334  }
335 
336  }
337  // DEBUGFN(4);
338 
339  std::sort(result.begin(), result.end(),
340  [center](uniqueRef<Finger>& a, uniqueRef<Finger>& b) {
341  return cv::norm(a.get()->lastPosition() - center) < cv::norm(b.get()->lastPosition() - center);
342  });
343  // DEBUGFN(5);
