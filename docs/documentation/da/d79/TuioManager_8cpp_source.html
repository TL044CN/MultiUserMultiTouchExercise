128 TuioCursor* TuioManager::addTuioCursor(float x, float y) {
129  sessionID++;
130 
131  int cursorID = (int)cursorList.size();
132  if ((int)(cursorList.size())<=maxCursorID) {
133  std::list<TuioCursor*>::iterator closestCursor = freeCursorList.begin();
134 
135  for(std::list<TuioCursor*>::iterator iter = freeCursorList.begin();iter!= freeCursorList.end(); iter++) {
136  if((*iter)->getDistance(x,y)<(*closestCursor)->getDistance(x,y)) closestCursor = iter;
137  }
138 
139  TuioCursor *freeCursor = (*closestCursor);
140  cursorID = (*closestCursor)->getCursorID();
141  freeCursorList.erase(closestCursor);
142  delete freeCursor;
143  } else maxCursorID = cursorID;
144 
145  TuioCursor *tcur = new TuioCursor(currentFrameTime, sessionID, cursorID, x, y);
146  cursorList.push_back(tcur);
147  updateCursor = true;
148 
149  for (std::list<TuioListener*>::iterator listener=listenerList.begin(); listener != listenerList.end(); listener++)
150  (*listener)->addTuioCursor(tcur);
151 
152  if (verbose /*&& tcur->getTuioState()!=TUIO_ADDED*/)
153  std::cout << "add cur " << tcur->getCursorID() << " (" << tcur->getSessionID() << ") " << tcur->getX() << " " << tcur->getY() << std::endl;
154 
155  return tcur;
156 }

201 void TuioManager::removeTuioCursor(TuioCursor *tcur) {
202  if (tcur==NULL) return;
203 
204  if (verbose /*&& tcur->getTuioState()!=TUIO_ADDED*/)
205  std::cout << "del cur " << tcur->getCursorID() << " (" << tcur->getSessionID() << ")" << std::endl;
206 
207  cursorList.remove(tcur);
208  tcur->remove(currentFrameTime);
209  updateCursor = true;
210 
211  for (std::list<TuioListener*>::iterator listener=listenerList.begin(); listener != listenerList.end(); listener++)
212  (*listener)->removeTuioCursor(tcur);
213 
214  if (tcur->getCursorID()==maxCursorID) {
215  maxCursorID = -1;
216  delete tcur;
217 
218  if (cursorList.size()>0) {
219  std::list<TuioCursor*>::iterator clist;
220  for (clist=cursorList.begin(); clist != cursorList.end(); clist++) {
221  int cursorID = (*clist)->getCursorID();
222  if (cursorID>maxCursorID) maxCursorID=cursorID;
223  }
224 
225  freeCursorBuffer.clear();
226  for (std::list<TuioCursor*>::iterator flist=freeCursorList.begin(); flist != freeCursorList.end(); flist++) {
227  TuioCursor *freeCursor = (*flist);
228  if (freeCursor->getCursorID()>maxCursorID) delete freeCursor;
229  else freeCursorBuffer.push_back(freeCursor);
230  }
231 
232  freeCursorList = freeCursorBuffer;
233 
234  } else {
235  for (std::list<TuioCursor*>::iterator flist=freeCursorList.begin(); flist != freeCursorList.end(); flist++) {
236  TuioCursor *freeCursor = (*flist);
237  delete freeCursor;
238  }
239  freeCursorList.clear();
240  }
241  } else if (tcur->getCursorID()<maxCursorID) {
242  freeCursorList.push_back(tcur);
243  }
244 }
