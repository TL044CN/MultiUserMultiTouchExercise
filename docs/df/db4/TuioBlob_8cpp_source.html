101 void TuioBlob::update (TuioTime ttime, float xp, float yp, float a, float w, float h, float f) {
102  TuioPoint lastPoint = path.back();
103  TuioContainer::update(ttime,xp,yp);
104 
105  TuioTime diffTime = currentTime - lastPoint.getTuioTime();
106  float dt = diffTime.getTotalMilliseconds()/1000.0f;
107  float last_rotation_speed = rotation_speed;
108 
109  float da = a-angle;
110  if (da > M_PI) da -= 2*M_PI;
111  else if (da < -M_PI) da+=2*M_PI;
112 
113  float prev_angle = angle;
114  if (angleFilter) angle = angleFilter->filter(angle+da,dt);
115  else angle = angle+da;
116  if (fabs(angle-prev_angle)<angleThreshold) angle = prev_angle;
117 
118  if (angle > 2*M_PI) angle-=2*M_PI;
119  else if (angle < 0) angle+=2*M_PI;
120 
121  da = angle-prev_angle;
122  if (da > M_PI) da -= 2*M_PI;
123  else if (da < -M_PI) da+=2*M_PI;
124  da = da/(2*M_PI);
125 
126  if (widthFilter && heightFilter) {
127  w = widthFilter->filter(w,dt);
128  h = heightFilter->filter(h,dt);
129  }
130 
131  float dw = fabs(width - w);
132  float dh = fabs(height - h);
133  if ((dw>sizeThreshold) || (dh>sizeThreshold)) {
134  width = w;
135  height = h;
136  }
137 
138  area = f;
139 
140  rotation_speed = (float)da/dt;
141  rotation_accel = (rotation_speed - last_rotation_speed)/dt;
142 
143  if ((rotation_accel!=0) && (state==TUIO_STOPPED)) state = TUIO_ROTATING;
144 }
