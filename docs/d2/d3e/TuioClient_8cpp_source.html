69 void TuioClient::processOSC( const ReceivedMessage& msg ) {
70  try {
71  ReceivedMessageArgumentStream args = msg.ArgumentStream();
72  //ReceivedMessage::const_iterator arg = msg.ArgumentsBegin();
73 
74  if( strcmp( msg.AddressPattern(), "/tuio/2Dobj" ) == 0 ){
75 
76  const char* cmd;
77  args >> cmd;
78 
79  if (strcmp(cmd,"source")==0) {
80  const char* src;
81  args >> src;
82 
83  source_name = strtok((char*)src, "@");
84  char *addr = strtok(NULL, "@");
85 
86  if (addr!=NULL) source_addr = addr;
87  else source_addr = (char*)"localhost";
88 
89  // check if we know that source
90  std::string source_str(src);
91  std::map<std::string,int>::iterator iter = sourceList.find(source_str);
92 
93  // add a new source
94  if (iter==sourceList.end()) {
95  source_id = (int)sourceList.size();
96  sourceList[source_str] = source_id;
97  } else {
98  // use the found source_id
99  source_id = sourceList[source_str];
100  }
101 
102  } else if (strcmp(cmd,"set")==0) {
103  int32 s_id, c_id;
104  float xpos, ypos, angle, xspeed, yspeed, rspeed, maccel, raccel;
105  args >> s_id >> c_id >> xpos >> ypos >> angle >> xspeed >> yspeed >> rspeed >> maccel >> raccel;
106 
107  lockObjectList();
108  std::list<TuioObject*>::iterator tobj;
109  for (tobj=objectList.begin(); tobj!= objectList.end(); tobj++)
110  if((*tobj)->getSessionID()==(long)s_id) break;
111 
112  if (tobj == objectList.end()) {
113 
114  TuioObject *addObject = new TuioObject((long)s_id,(int)c_id,xpos,ypos,angle);
115  frameObjects.push_back(addObject);
116 
117  } else if ( ((*tobj)->getX()!=xpos) || ((*tobj)->getY()!=ypos) || ((*tobj)->getAngle()!=angle) || ((*tobj)->getXSpeed()!=xspeed) || ((*tobj)->getYSpeed()!=yspeed) || ((*tobj)->getRotationSpeed()!=rspeed) || ((*tobj)->getMotionAccel()!=maccel) || ((*tobj)->getRotationAccel()!=raccel) ) {
118 
119  TuioObject *updateObject = new TuioObject((long)s_id,(*tobj)->getSymbolID(),xpos,ypos,angle);
120  updateObject->update(xpos,ypos,angle,xspeed,yspeed,rspeed,maccel,raccel);
121  frameObjects.push_back(updateObject);
122 
123  }
124  unlockObjectList();

126  } else if (strcmp(cmd,"alive")==0) {

129  aliveObjectList.clear();
130  while(!args.Eos()) {

132  aliveObjectList.push_back((long)s_id);

141  if (fseq>currentFrame) currentTime = TuioTime::getSessionTime();
142  if ((fseq>=currentFrame) || ((currentFrame-fseq)>100)) currentFrame = fseq;
143  else lateFrame = true;
144  } else if ((TuioTime::getSessionTime().getTotalMilliseconds()-currentTime.getTotalMilliseconds())>100) {
145  currentTime = TuioTime::getSessionTime();
146  }

148  if (!lateFrame) {
149 
150  lockObjectList();
151  //find the removed objects first
152  for (std::list<TuioObject*>::iterator tobj=objectList.begin(); tobj != objectList.end(); tobj++) {
153  if ((*tobj)->getTuioSourceID()==source_id) {
154  std::list<long>::iterator iter = find(aliveObjectList.begin(), aliveObjectList.end(), (*tobj)->getSessionID());
155  if (iter == aliveObjectList.end()) {
156  (*tobj)->remove(currentTime);
157  frameObjects.push_back(*tobj);
158  }
159  }
160  }
161  unlockObjectList();
162 
163  for (std::list<TuioObject*>::iterator iter=frameObjects.begin(); iter != frameObjects.end(); iter++) {
164  TuioObject *tobj = (*iter);
165 
166  TuioObject *frameObject = NULL;
167  switch (tobj->getTuioState()) {
168  case TUIO_REMOVED:
169 
170  frameObject = tobj;
171  frameObject->remove(currentTime);
172 
173  for (std::list<TuioListener*>::iterator listener=listenerList.begin(); listener != listenerList.end(); listener++)
174  (*listener)->removeTuioObject(frameObject);
175 
176  lockObjectList();
177  for (std::list<TuioObject*>::iterator delobj=objectList.begin(); delobj!=objectList.end(); delobj++) {
178  if((*delobj)->getSessionID()==frameObject->getSessionID()) {
179  objectList.erase(delobj);
180  break;
181  }
182  }
183  unlockObjectList();
184  break;
185  case TUIO_ADDED:
186 
187  lockObjectList();
188  frameObject = new TuioObject(currentTime,tobj->getSessionID(),tobj->getSymbolID(),tobj->getX(),tobj->getY(),tobj->getAngle());
189  if (source_name) frameObject->setTuioSource(source_id,source_name,source_addr);
190  objectList.push_back(frameObject);
191  unlockObjectList();
192 
193  for (std::list<TuioListener*>::iterator listener=listenerList.begin(); listener != listenerList.end(); listener++)
194  (*listener)->addTuioObject(frameObject);
195 
196  break;
197  default:
198 
199  lockObjectList();
200  std::list<TuioObject*>::iterator iter;
201  for (iter=objectList.begin(); iter != objectList.end(); iter++) {
202  if (((*iter)->getTuioSourceID()==source_id) && ((*iter)->getSessionID()==tobj->getSessionID())) {
203  frameObject = (*iter);
204  break;
205  }
206  }
207 
208  if (iter==objectList.end()) {
209  unlockObjectList();
210  break;
211  }

214  frameObject->update(currentTime,tobj->getX(),tobj->getY(),tobj->getAngle());
215  else
216  frameObject->update(currentTime,tobj->getX(),tobj->getY(),tobj->getAngle(),tobj->getXSpeed(),tobj->getYSpeed(),tobj->getRotationSpeed(),tobj->getMotionAccel(),tobj->getRotationAccel());
217 
218  unlockObjectList();

221  (*listener)->updateTuioObject(frameObject);
222  }

227  (*listener)->refresh(currentTime);

231  TuioObject *tobj = (*iter);
232  delete tobj;
233  }
