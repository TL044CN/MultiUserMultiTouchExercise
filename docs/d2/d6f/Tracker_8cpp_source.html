28 void Tracker::trackFingers(const cv::Mat& frame) {
29 
30  //clean up fingers that are older than 5 frames
31  for ( auto maybeFingerRef : mFingers ) {
32  DEBUGFNEX("Finger? %i", maybeFingerRef.has_value());
33  if ( !maybeFingerRef.has_value() ) continue;
34  auto& finger = maybeFingerRef->get();
35 
36  if ( !finger ) continue;
37 
38  DEBUGFNEX("id[%lu] coord[%f,%f] age[%i] ",
39  finger->id(), finger->lastPosition().x, finger->lastPosition().y, finger->getAge());
40  finger->incrementAge();
41  if ( finger->getAge() > ceMaxAge ) mFingers.remove(finger);
42  }
43 
44  Mat maskedImg = frame - mBackgroundImage;
45  Mat tempImg1, tempImg2, processedImg;
46 
47  blur(maskedImg, tempImg2, Size2i(20, 20));
48  tempImg1 = maskedImg - tempImg2;
49  blur(tempImg1, tempImg2, Size2i(5, 5));
50  threshold(tempImg2, tempImg1, 6, 255, cv::THRESH_BINARY);
51  blur(tempImg1, processedImg, Size2i(5, 5));
52 
53  std::vector<std::vector<Point>> contours;
54  std::vector<Vec4i> hierarchy;
55  std::vector<RotatedRect> ellipses;
56 
57  findContours(processedImg, contours, hierarchy, cv::RETR_CCOMP, cv::CHAIN_APPROX_SIMPLE);
58 
59  if ( hierarchy.size() <= 0 ) return;
60  for ( int idx = 0; idx >= 0; idx = hierarchy[idx][0] ) {
61  if ( contourArea(cv::Mat(contours[idx])) > 30 && contours[idx].size() > 4
62  ) {
63  auto ellipse = cv::fitEllipse(Mat(contours[idx]));
64 
65  ellipses.push_back(ellipse);
66  }
67  }
68 
69  if ( ellipses.empty() ) return;
70 
71  DEBUGFNEX("Found %lu Ellipses. Trying to fit to Fingers", ellipses.size());
72 
73  for ( auto& ellipse : ellipses ) {
74  if ( mFingers.empty() ) {
75  DEBUGFNEX("First Ellipse. Insert directly.\n");
76  mFingers.insert(std::make_unique<Finger>(ellipse));
77  break;
78  }
79 
80  DEBUGFNEX("Finding Candidates for Ellipse at [%f , %f]", ellipse.center.x, ellipse.center.y);
81  auto candidates = mFingers.intersect(ellipse.center, ceMaxFingerDist);
82 
83  if ( candidates.size() <= 0 ) {
84  DEBUGFNEX("no Candidates found. Insert directly.\n");
85  mFingers.insert(std::make_unique<Finger>(ellipse));
86  } else {
87  DEBUGFNEX("Candidates found. Check for match");
88  bool found = false;
89  for ( auto& candidate : candidates ) {
90  if ( !candidate.get() ) continue;
91  DEBUGFNEX("Finger Found id[%lu] age[%i] pos[%f , %f]",
92  candidate.get()->id(),
93  candidate.get()->getAge(),
94  candidate.get()->lastPosition().x, candidate.get()->lastPosition().y
95  );
96 
97  if ( candidate.get()->getAge() == 0 ) continue;
98 
99  mFingers.update(candidate, ellipse);
100  found = true;
101  break;
102  }
103  if ( !found ) {
104  DEBUGFNEX("Finger is new. Insert directly\n");
105  mFingers.insert(std::make_unique<Finger>(ellipse));
106  } else {
107  DEBUGFNEX("Finger found and updated.\n");
108  }
109  }

111  }
